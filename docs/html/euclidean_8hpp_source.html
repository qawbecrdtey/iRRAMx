198 // for any hypercube H of size 2^-q with corners aligned by 2^-q in hypercube H',
199 // f(H) is subset of a hypercube of size 2^-p
200 // f: R^M -> R^N
201 // H' is given with the center c and side length 2^-s
202 template<int M, int N>
203 int module2_(std::function<Point<N>(Point<M>)> f, int p, const HyperCube<M> &c, int s) {
204  sizetype err;
205 
206  // create H
207  HyperCube<M> box = c;
208  for(REAL &u : box) {
209  sizetype_set(err, 1, -s-1);
210  u.seterror(err);
211  }
212 
213  // check if f(H') is subset of a hypercube
214  DYADIC d;
215  bool success = true;
216  try {
217  single_valued code;
218 
219  HyperCube<N> fBox = f(box);
220  for(REAL &v : fBox) d = approx(v, -p-1);
221  } catch (Iteration it) { success = false; }
222 
223  // return current one if success
224  if(success) return s;
225 
226  // bisection on failure, total 2^M sub-hypercubes
227  int result = s+1;
228  HyperCube<M> offset, newCenter;
229  REAL halfLen = Exp(-s-1);
230  offset.fill(0);
231  for(int k=0 ; k<(1<<M) ; k++) {
232  // configure offset
233  // If the j'th bit of k is 0, we choose the left half section on j'th dimension
234  // Otherwise, the right half section.
235  for(int j=0 ; j<M ; j++) {
236  int sign = ((k & (1<<j))<<1)-1; // -1 on left section, 1 otherwise
237  offset[j] = REAL(INTEGER(sign)) * halfLen;
238  }
239 
240  // configure point

242 
243  // find the required precision
244  result = max(result, module2_<M,N>(f, p, newCenter, s+1));
245  }
